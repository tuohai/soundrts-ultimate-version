; 建筑选择界面热键
; 在 style.txt 为单位设置 keyboard building1 … keyboard building16（与 keyboard building 并存）
;
; 键位（与士兵界面同区，两行各 8 个）：
;   d f g h j k l ;  → building1 – building8
;   e r t y u i o p  → building9 – building16
; 每键：本键选同类建筑中的一个，SHIFT+本键逐个切换，CTRL+本键选全图同类

#define BUILDING building1 building2 building3 building4 building5 building6 building7 building8 building9 building10 building11 building12 building13 building14 building15 building16

; 通用 building 类型（未指定 buildingN 的建筑）
q: select_unit 1 building even_if_no_menu
SHIFT q: select_unit -1 building even_if_no_menu

BACKSLASH: command_unit

; 过滤器（m-n，选目标时）
m: toggle_side_filter 1
SHIFT m: toggle_side_filter -1
n: toggle_type_filter 1
SHIFT n: toggle_type_filter -1
